select  A.TABLE_NAME, D.COMMENTS
	, A.COLUMN_NAME
	, A.DATA_TYPE || '(' || A.DATA_LENGTH || ')' TYPE
	, C.COMMENTS
	, A.NULLABLE
	, NVL2(B.COLUMN_NAME, 'PK', '')
from    all_tab_cols A
	,(select S.TABLE_NAME, C.COLUMN_NAME from USER_CONS_COLUMNS C, USER_CONSTRAINTS S
	where   C.CONSTRAINT_NAME = S.CONSTRAINT_NAME and S.CONSTRAINT_TYPE = 'P') B, USER_COL_COMMENTS C, ALL_TAB_COMMENTS D
where   1 = 1
		and     A.OWNER         =   'NHIS3'
		and		A.TABLE_NAME	=  C.TABLE_NAME(+)
		and		A.TABLE_NAME	=  D.TABLE_NAME(+)
		and		A.COLUMN_NAME	=  C.COLUMN_NAME(+)
		and     A.TABLE_NAME    =   B.TABLE_NAME(+)
		and     A.COLUMN_NAME   =   B.COLUMN_NAME(+)
order by A.TABLE_NAME, A.COLUMN_ID
